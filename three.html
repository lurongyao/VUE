js:

//1. 准备一个根组件

var App=Vue.extend\(\);



//2. Home News组件都准备

var Home=Vue.extend\({

    template:'&lt;h3&gt;我是主页&lt;/h3&gt;'

}\);



var News=Vue.extend\({

    template:'&lt;h3&gt;我是新闻&lt;/h3&gt;'

}\);



//3. 准备路由

var router=new VueRouter\(\);



//4. 关联

router.map\({

    'home':{

        component:Home

    },

    'news':{

        component:News

    }

}\);



//5. 启动路由

router.start\(App,'\#box'\);

跳转:

router.redirect\({

    ‘/’:'/home'

}\);
